async function loadInsights() { const container = document.getElementById('insights-container'); if (!container) return; try { // Fetch the pre-built JSON file directly // Using a timestamp query param to prevent browser caching of old data const res = await fetch('/insights.json?v=' + new Date().getTime()); if (!res.ok) throw new Error("Index file not found"); const posts = await res.json(); const htmlContent = posts.map(post => { // Date formatting const dateObj = new Date(post.date); const displayDate = dateObj.toLocaleDateString('en-US', { month: 'short', year: 'numeric' }).toUpperCase(); return `
${displayDate} / ${post.is_external ? 'DISPATCH' : 'INSIGHT'}

${post.title} ${post.is_external ? 'â†—' : ''}

${post.description}

`; }).join(''); container.innerHTML = htmlContent; } catch (e) { console.error("Load Error:", e); container.innerHTML = '

Unable to load insights.

'; } } window.onload = loadInsights;
